Function$ getword{word,n}
 if n>1 then
  for t = 1 to n
   sp = Instr(word," ")
   if sp = 0 then
    pop for
    goto done
   endif
  next
 else
  fs = 1
 endif

 sp + 1

 sp2 = Instr(word," ",sp)
 if sp2 = 0 then goto done

 subword = Mid$(word,sp,sp2-sp-1)

done:
 Function Return subword
EndFunction
