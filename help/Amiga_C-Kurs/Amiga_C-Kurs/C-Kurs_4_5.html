Zeiger als Rückgabewerte


Zeiger können auch als Rückgabewert einer Funktion auftreten.

Das folgende Beispielprogramm ist eine Abwandlung von Listing 4.4.1:

/*  Listing 4.5.1
 *  Zeiger als Rückgabewert einer Funktion
 */
#include <stdio.h>
#include <stdlib.h>  // Wird für malloc benötigt
// Struktur für Adresse
 struct Adresse
{
   char Vorname[20];
   char Nachname[20];
   char Strasse[40];
   char Plz[6];
   char Stadt[20];
};

/*  Erzeuge_Adresse
 *
 *  Fordert den Benutzer zur Eingabe der Adresse auf und gibt
 *  einen Zeiger auf eine Struktur vom Typ Adresse zurück.
 *
 *  Eingabeparameter: keine
 *  Rückgabe: Zeiger auf Struktur vom Typ Adresse
 */
 struct Adresse *Erzeuge_Adresse(void)
{
  // Ein Zeiger auf eine Struktur vom Typ Adresse
   struct Adresse *a;

  // Speicher für Struktur beschaffen
   a = (struct Adresse *)malloc(sizeof(struct Adresse));

  // Daten für Anschrift einlesen
   printf("Bitte geben Sie Ihre Adresse ein:\n");
  printf("Vornamen: ");
  gets(a->Vorname);
  printf("Nachnamen: ");
  gets(a->Nachname);
  printf("Straße und Hausnummer: ");
  gets(a->Strasse);
  printf("Postleitzahl: ");
  gets(a->Plz);
  printf("Stadt: ");
  gets(a->Stadt);

  // Zeiger auf Struktur vom Typ Adresse zurückgeben
   return a;

}

/*  Zeige_Adresse
 *  Gibt eine Adresse aus
 *
 *  Eingabeparameter:
 *  struct Adresse *a : Zeiger auf Strukturvariable vom Typ Adresse
 *
 */
 void Zeige_Adresse(struct Adresse *a)
{
   printf("%s %s\n",a->Vorname,a->Nachname);
   printf("%s\n",a->Strasse);
   printf("%s %s\n",a->Plz,a->Stadt);
}

int main(void)
{
  // Ein Zeiger auf eine Strukturvariable vom Typ Adresse
   struct Adresse *Zeiger;

  // Adresse mittels Funktionsaufruf erzeugen
   Zeiger = Erzeuge_Adresse();

  printf("\n");

  // Anschrift ausgeben
   Zeige_Adresse(Zeiger);

  return 0;
}

Der Unterschied zu Listing 4.4.1 ist, daß hier die Daten für die Adresse nicht innerhalb von main, sondern in einer extra Funktion Namens Erzeuge_Adresse eingelesen werden. Der Rückgabewert dieser Funktion ist ein Zeiger auf eine Struktur von Typ Adresse.

Innerhalb der Funktion Erzeuge_Adresse wird zunächst ein Zeiger auf eine Struktur vom Typ Adresse deklariert:

  // Ein Zeiger auf eine Struktur vom Typ Adresse
   struct Adresse *a;

Anschließend wird Speicher für eine Struktur vom Typ Adresse beschafft:

  // Speicher für Struktur beschaffen
   a = (struct Adresse *)malloc(sizeof(struct Adresse));

Die Funktion malloc fordert Speicher der angegebenen Größe an und gibt einen Zeiger auf die entsprechende Speicherstelle zurück. Um zu ermitteln, wieviel Speicher benötigt wird, benutzen wir den sizeof-Operator. Da malloc einen Zeiger vom Typ void zurückgibt, wir aber einen Zeiger auf eine Struktur vom Typ Adresse benötigen, müßen wir mit (struct Adresse *) noch eine explizite Typumwandlung vornehmen.

Weil der Funktionsprototyp von malloc in der Include-Datei stdlib.h definiert ist, muß diese am Anfang des Programms eingebunden werden:

#include <stdlib.h>  // Wird für malloc benötigt

Nachdem alle Komponenten der Struktur ausgefüllt sind, wird der Zeiger a mittels return-Anweisung an die aufrufende Funktion zurückgegeben:

  // Zeiger auf Struktur vom Typ Adresse zurückgeben
   return a;

Innerhalb von main können wir dann den Rückgabewert der Funktion Erzeuge_Adresse unserer Zeigervariablen Zeiger zuweisen:

  // Adresse mittels Funktionsaufruf erzeugen
   Zeiger = Erzeuge_Adresse();

Um die Thematik zu vertiefen, wollen wir uns nun mit Vektorrechnen beschäftigen. Das Rechnen mit Vektoren ist z.B. Grundlage für physikalische Berechnungen und 3D-Grafik.

/*  Vektor3D.h

#ifndef VEKTOR3D_H

 *  Erzeugt 3D Vektor

 *  Rückgabe: struct Vektor3D* : Zeiger auf Struktur vom Typ Vektor3D

Hier das Implementierungsmodul:

/*  Vektor3D.c
 *  Implementierungsmodul für Vektorrechnen
 */
#include "Vektor3D.h"

/*  ErzeugeVektor3D
 *
 *  Erzeugt 3D Vektor
 *
 *  Eingabeparameter: x,y,z : Komponenten des Vektors
 *  Rückgabe: struct Vektor3D* : Zeiger auf Struktur vom Typ Vektor3D
 *
 */
 struct Vektor3D *ErzeugeVektor3D(double x, double y, double z)
{
   struct Vektor3D *v;

   // Speicher für Vektor3D-Struktur beschaffen
    v = (struct Vektor3D *)malloc(sizeof(struct Vektor3D));

   // Komponenten ausfüllen
    v->x = x;
   v->y = y;
   v->z = z;

   return v;
}

/*  ZeigeVektor3D
 *  
 *  Vektor3D-Struktur auf der Konsole ausgeben
 *
 *  Eingabeparameter: struct Vektor3D *v : Zeiger auf Vektor3D-Struktur
 *  Rückgabe: keine
 *
 */
 void ZeigeVektor3D(struct Vektor3D *v)
{
   printf("(%lf,%lf,%lf)",v->x,v->y,v->z);
}

/*  Vector3DAdd
 *
 *  Zwei dreidimensionale Vektoren addieren
 *
 *  Eingabeparameter: struct Vektor3D *v1 : Zeiger auf ersten Vektor
 *                    struct Vektor3D *v2 : Zeiger auf zweiten Vektor
 *  Rückgabe: struct Vektor3D* : Summe von v1 und v2
 *
 */
 struct Vektor3D *Vektor3DAdd(struct Vektor3D *v1, struct Vektor3D *v2)
{
   return ErzeugeVektor3D(v1->x + v2->x, v1->y + v2-> y, v1->z + v2->z);
}

/*  Vector3DSub
 *
 *  Zwei dreidimensionale Vektoren subtrahieren
 *
 *  Eingabeparameter: struct Vektor3D *v1 : Zeiger auf ersten Vektor
 *                    struct Vektor3D *v2 : Zeiger auf zweiten Vektor
 *  Rückgabe: struct Vektor3D* : Differenz von v1 und v2
 *
 */
 struct Vektor3D *Vektor3DSub(struct Vektor3D *v1, struct Vektor3D *v2)
{
   return ErzeugeVektor3D(v1->x - v2->x, v1->y - v2-> y, v1->z - v2->z);
}

/*  Vector3DMult
 *
 *  Einen Vektor mit einem Skalar multiplizieren
 *
 *  Eingabeparameter: struct Vektor3D *v1 : Zeiger auf ersten Vektor
 *                    double s : Ein Skalar
 *  Rückgabe: struct Vektor3D* : v1 multipliziert mit Skalar s
 *
 */
 struct Vektor3D *Vektor3DMult(struct Vektor3D *v1, double s)
{
   return ErzeugeVektor3D(v1->x * s, v1->y * s, v1->z * s);
}

/*  Vector3DSMult
 *
 *  Skalarprodukt zweier 3D-Vektoren
 *
 *  Eingabeparameter: struct Vektor3D *v1 : Zeiger auf ersten Vektor
 *                    struct Vektor3D *v2 : Zeiger auf zweiten Vektor
 *  Rückgabe: double : Skalarprodukt von v1 und v2
 *
 */
 double Vektor3DSMult(struct Vektor3D *v1, struct Vektor3D *v2)
{
   return (v1->x * v2->x + v1->y * v2-> y + v1->z * v2->z);
}

Die Funktion ErzeugeVektor3D erzeugt eine neue Struktur vom Typ Vektor3D, trägt die Komponenten x, y und z ein. Die Werte für x, y und z erhält ErzeugeVektor3D über Funktionsparameter. Der Rückgabewert ist ein Zeiger auf eine Struktur vom Typ Vektor3D.

Folgende mathematische Notation

können wir damit als C-Code wir folgt formulieren:

   v1 = ErzeugeVektor3D(1.5,-0.5,2.5);

Die Funktionen für Vektoraddition, Vektorsubtraktion und der Multiplikation eins Vektors mit einem Skalar bauen auf diese Funktion auf und liefern jeweils einen Zeiger auf eine neue Struktur vom Typ Vektor3D zurück, die das Ergebnis der jeweiligen Rechnung enthält. So wird z.B. in der Funktion Vektor3DAdd die Funktion ErzeugeVektor3D wie folgt aufgerufen:

 ErzeugeVektor3D(v1->x + v2->x, v1->y + v2-> y, v1->z + v2->z);

Die x-Komponente des neuen Vektors ist damit das Ergebnis der Addition der x-Komponente des Vektors v1 und der x-Komponente des Vektors v2.

 v1->x + v2->x

Auch die y und z-Komponenten des neuen Vektors ergeben sich aus Addition der Komponenten der Vektoren v1 un v2.

Die Funktion Vektor3DSMult berechnet des Skalarprodukt zweier 3D-Vektoren und liefert ein Skalar als double-Wert. ZeigeVektor3D dient dazu, einen 3D-Vektor mit Hilfe von printf auf der Konsole auszugeben.

Mit Listing 4.5.2 testen wir unsere Module:

/*  Listing 4.5.2
 *  Vektorrechnen
 */
#include <stdio.h>

#include "Vektor3D.h"
#include "Vektor3D.c"

int main(void)
{
   // Zwei Zeiger auf Vektor3D-Strukturen
    struct Vektor3D *v1,*v2;
   double s = -0.5;  // Ein Skalar
   // Drei 3D-Vektoren erzeugen
    v1 = ErzeugeVektor3D(1.5,-0.5,2.5);
   v2 = ErzeugeVektor3D(0.5,1.5,2.5);

   // Vektoren auf der Konsole ausgeben
    printf("v1 = ");
   ZeigeVektor3D(v1);
   printf("\n");
   printf("v2 = ");
   ZeigeVektor3D(v2);
   printf("\n");

   // Vektoraddition: v1 + v2
    printf("v1 + v2 = ");
   ZeigeVektor3D(Vektor3DAdd(v1,v2));
   printf("\n");

   // Vektorsubtraktion: v1 - v2
    printf("v1 - v2 = ");
   ZeigeVektor3D(Vektor3DSub(v1,v2));
   printf("\n");

   // Multiplikation mit einem Skalar: v1 * s;
    printf("v1 * %lf = ",s);
   ZeigeVektor3D(Vektor3DMult(v1,s));
   printf("\n");

   // Skalarmultiplikation: s = v1 * v2
    printf("v1 * v2 = %lf\n",Vektor3DSMult(v1,v2));

   return 0;
}
