Mit dem Blitter arbeiten


Als der erste Amiga vor 20 Jahren erschien, beeindruckte er vor allem durch seine im Gegensatz zu anderen Computern dieser Zeit phänomenalen Grafikfähigkeiten. Diese konnten damals nur durch spezielle Chips - die sogenannten Custom Chips realisiert werden. Einer dieser Chips ist der Blitter. Er ermöglich unabhängig von der CPU ganze Speicherbereiche in einem Rutsch zu kopieren - und das blitzschnell.

So kann man mit dem Blitter zum Beispiel blitzschnell einen Ausschnitt des Bildschirms an eine andere Stelle kopieren, ohne daß die CPU dabei in die Knie geht.

Die graphics.library stellt dazu diverse Funktionen zur Verfügung. In dieser Lektion wollen wir die Funktion ClipBlit betrachten. Diese wird dazu verwendet, um einen Ausschnitt eines Fensters zu kopieren.

ClipBlit

Syntax: 

  ClipBlit(Src, SrcX, SrcX, Dest, DestX, DestY, XSize, YSize, Minterm);

Kopiert einen rechteckigen Bereich eines RastPorts.

Eingabeparameter:

Src : Zeiger auf den RastPort der Quelle
SrcX, SrcY : Koordinaten des linken, oberen Eckpunkts des Quellbereichs
Dest : Zeiger auf den RastPort des Ziels
DestX, DestY : Koordinaten des linken, oberen Eckpunkts des Zielbereichs
XSize,YSize : Höhe und Breite des zu kopierenden Ausschnitts. Muß mindestens 1 sein.
Minterm : Legt fest, wie die Kopie erfolgen soll. 0C0 bedeutet 1:1.

Die Funktion ClipBlit hat folgende Besonderheit: Sie macht Clipping. Das bedeutet, daß Teile der Grafik, die nicht in den Ziel RastPort passen einfach abgeschnitten werden.

Mittlerweile haben auch moderne Grafikkarten mit dem Amiga Blitter vergleichbare Schaltungen. Den Blitter-Funktionen der graphics.library ist es egal welche Grafikhardware Sie verwenden. Ob OCS, ECS, AGA oder eine Grafikkarte - das Amiga Betriebssystem entscheidet selbständig, welche Hardware zum Blitten verwendet wird.

Listing 8.9.1 demonstriert die Verwendung der Funktion ClipBlit:

/*  Listing 8.9.1
 *  Mit dem Blitter arbeiten
 *
 *  Dieses Beispielprogramm zeichnet zunächst eine kleine Fraktalgrafik
 *  im linken, oberen Eck des Bildschirms. Anschließend wird der Ausschnitt
 *  mit Hilfe des Blitters auf den ganzen Bildschirm kopiert.
 *
 */
#include <exec/types.h>
#include <intuition/intuition.h>
#include <graphics/gfx.h>

#include <proto/intuition.h>
#include <proto/graphics.h>
#include <proto/dos.h>
#include <proto/exec.h>

struct IntuitionBase  *IntuitionBase = NULL;
struct GfxBase        *GfxBase = NULL;
struct Screen         *Bildschirm = NULL;           // Zeiger auf Screen Struktur
 struct Window         *Fenster = NULL;
struct RastPort       *rp;

/* Funktionsprototypen */
 void SetzeFarben(struct ViewPort *vp);
void ZeichneFraktal(struct RastPort *rp, int xc, int yc, int xw, int yh, int z, int t);
void FuelleBildschirm(struct RastPort *rp,int w, int h,int g, int z, int t);

int main(void)
{
   int width,height;
   struct ViewPort *vp;

   // Öffne Intuition Library
    IntuitionBase = (struct IntuitionBase *) OpenLibrary("intuition.library", 37L);
   GfxBase = (struct GfxBase *) OpenLibrary("graphics.library", 37L);

   if (IntuitionBase != NULL)
   {

     /* Öffne neuen Screen */
      Bildschirm = OpenScreenTags(NULL,
                                 SA_LikeWorkbench, TRUE,   // Soll wie WB Screen sein
                                  TAG_DONE);

     if (Bildschirm != NULL)
     {
       /*  Wir öffnen ein rahmenloses Fenster, welches den gesammten
        *  Screen ausfüllt
        */
       /* Höhe und Breite vom Screen */
        width = Bildschirm->Width;
       height = Bildschirm->Height;

       Fenster = OpenWindowTags(NULL,
                                WA_Width, width,
                                WA_Height, height,
                                WA_Borderless, TRUE,          // Rahmenlos
                                 WA_CustomScreen, Bildschirm,  // Auf welchen Screen?
                                 WA_RMBTrap, TRUE,             // Rechten Maus Knopf abfangen
                                 WA_Activate, TRUE,
                                TAG_DONE);


       vp = ViewPortAddress(Fenster);
       SetzeFarben(vp);

       rp = Fenster->RPort;

       /* Grafik zeichnen */
        FuelleBildschirm(rp,width,height,64,398,2);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,16,5,6);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,212,5);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,11,5);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,100,4);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,16,17,6);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,1111,3);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,124,3);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,8,1);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,64,641,1);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,64,6,2);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,9,2);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,32,24,3);
       Delay(150);  // Warte 3 Sekunden
        FuelleBildschirm(rp,width,height,128,5094,3);
       Delay(150);  // Warte 3 Sekunden
     } // end if
     Delay(150);               // Warte 3 Sekunden
     if (Fenster != NULL) CloseWindow(Fenster);

     if (Bildschirm != NULL)   // Falls der Screen geöffnet werden konnte...
      CloseScreen(Bildschirm);  // Schließe Screen
   } // end if
   if (GfxBase != NULL) CloseLibrary((struct Library *) GfxBase);
   if (IntuitionBase != NULL) CloseLibrary((struct Library *) IntuitionBase);

   return 0;
}

/*  SetzeFarben
 *
 *  Setzt die Farbwerte für die ersten 32 Farben (0-31) eines ViewPorts
 *
 */
 void SetzeFarben(struct ViewPort *vp)
{
   int x,c;
   SHORT color;
   UBYTE r,g,b;

   color=0;

   for (x=0;x<=15;x++)
   {
     color = (SHORT)(x*256+x*16+x*0.8);
     r=color >> 8 & 0xF;
     g=color >> 4 & 0xF;
     b=color & 0xF;
     SetRGB4(vp,x,r,g,b);
   }

   for (x=16;x<=31;x++)
   {
     c = 31-x;
     color = (SHORT)(c*256+c*16+c*0.8);
     r=color >> 8 & 0xF;
     g=color >> 4 & 0xF;
     b=color & 0xF;
     SetRGB4(vp,x,r,g,b);
   }

}

/*  ZeichneFraktal
 *
 *  Eingabeparameter:
 *
 *  struct RastPort *rp : Der RastPort, in den gezeichnet werden soll
 *  int xc, yc          : Koordinaten des Zentrums
 *  int xw, yh          : Breite und Höhe des Fraktals
 *  int z               : Zoomfaktor
 *  int t               : Typ des Fraktals
 *
 */
 void ZeichneFraktal(struct RastPort *rp, int xc, int yc, int xw, int yh, int z, int t)
{
  int x,y,xx,yy;
  xx = 0;
  yy = 0;

  switch (t) // Welcher Fraktaltyp?
   {
    case 1:

    for (y=-yh;y<=yh;y++)
    {
      for (x=-xw;x<=xw;x++)
      {
         SetAPen(rp,((x*x+y*y)*z/100)&31);
         WritePixel(rp,xc+x,yc+y);
      }
    }

    break;

    case 2:

    for (y=-yh;y<=yh;y++)
    {
      for (x=-xw;x<=xw;x++)
      {
         SetAPen(rp,((x*x-y*y)*z/100)&31);
         WritePixel(rp,xc+x,yc+y);
      }
    }

    break;

    case 3:

    for (y=-yh;y<=yh;y++)
    {
      for (x=-xw;x<=xw;x++)
      {
         xx=xx+x-(int)(x/10)*10;
         SetAPen(rp,((xx+yy)*z/100)&31);
         WritePixel(rp,xc+x,yc+y);
      }
      yy=yy+y-(int)(y/10)*10;
    }

    break;

    case 4:

    for (y=-yh;y<=yh;y++)
    {
      for (x=-xw;x<=xw;x++)
      {
         SetAPen(rp,((x*x*y+x*y*y)*z/1000)&31);
         WritePixel(rp,xc+x,yc+y);
      }
    }

    break;

    case 5:

    for (y=-yh;y<=yh;y++)
    {
      for (x=-xw;x<=xw;x++)
      {
         SetAPen(rp,((x*x*y*y)/z)&31);
         WritePixel(rp,xc+x,yc+y);
      }
    }

    break;

    case 6:

    for (y=-yh;y<=yh;y++)
    {
      for (x=-xw;x<=xw;x++)
      {
         SetAPen(rp,((x*y*z)&31));
         WritePixel(rp,xc+x,yc+y);
      }
    }

    break;

  } // Ende der switch Verzweigung
}

void FuelleBildschirm(struct RastPort *rp,int w, int h,int g, int z, int t)
{
  int x,y,gz;

  SetRast(rp,0L);
  /* Zunächst wird eine "Kachel" erzeugt... */
   ZeichneFraktal(rp,g,g,g,g,z,t);
  gz=g*2-1;

  /* Fülle den gesamten Bildschirm mit Hilfe des Blitters */
   for (x=0;x<=w/(g*2);x++)
  {
     for (y=0;y<=h/(g*2);y++)
     {
        ClipBlit(rp,0,0,rp,x*gz,y*gz,gz,gz,0x0c0);
     }
  }
}

Die Funktion FuelleBildschirm füllt den gesamten Bildschirm mit Grafik. Dazu wird zunächst in der linken, oberen Ecke mit Hilfe der Funktion ZeichneFraktal eine kleine Fraktalgrafik gezeichnet.


In einer for-Schleife wird diese dann mit ClipBilit auf den gesamten Bildschirm kopiert. Stellen Sie sich die kleine Fraktalgrafik einfach wie eine Kachel (englische Bezeichnung: Tile) vor. Das gesamte Bild entsteht sozusagen durch nahtloses Aneinanderfügen solcher Kacheln. Diese Technik wird auch häufig in Spielen verwendet, wenn man aus immer wiederkehrenden Elementen eine koplexe Grafik erzeugen will. Man spricht dann auch von Tile-Grafik.
